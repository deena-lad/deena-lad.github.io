Calculate and filter by percentiles

def calculate_percentiles(data_array, lower_per, upper_per):
    
    valid_data = data_array.values[~np.isnan(data_array.values)]
    
    if len(valid_data) == 0:
        return 0, 0
    
    lp = np.percentile(valid_data, lower_per)
    up = np.percentile(valid_data, upper_per)
    
    # print(f"{lower_per} percentile: {lp:.2e}")
    # print(f"{upper_per} percentile: {up:.2e}")
    
    return lp, up
def filter_data_by_percentiles(data_array, lp, up):

    filtered_data = data_array.copy()
    filtered_data = filtered_data.where((filtered_data >= lp) & (filtered_data <= up))
    
    return filtered_data

Prepare data for region

def prepare_data_for_region(data_array, region='world', shapefile_path=None, region_bounds=None):

    if region == 'world':
        print("Using global data extent")
        return data_array
    
    elif region == 'india':
        if region_bounds is None:
            min_lat, max_lat = 8, 38
            min_lon, max_lon = 65, 98
        else:
            min_lon, max_lon, min_lat, max_lat = region_bounds
        
        # Crop data to India bounds
        cropped_data = data_array.sel(lat=slice(min_lat, max_lat), lon=slice(min_lon, max_lon))

        # ds_2019 = ds_2019.sel(lat=slice(min_lat, max_lat), lon=slice(min_lon, max_lon))
        
        print(f"Data cropped to India: {min_lon}°E-{max_lon}°E, {min_lat}°N-{max_lat}°N")
        print(f"Shape: {data_array.shape} → {cropped_data.shape}")
        
        # Apply shapefile mask if provided
        if shapefile_path:
            try:
                gdf = gpd.read_file(shapefile_path)
                print(f"Shapefile loaded: {len(gdf)} features")
                
                # Create mask using shapefile
                lons, lats = np.meshgrid(cropped_data.lon.values, cropped_data.lat.values)
                points = [Point(lon, lat) for lon, lat in zip(lons.flatten(), lats.flatten())]
                
                mask = np.zeros(len(points), dtype=bool)
                boundary_geom = gdf.unary_union
                
                for i, point in enumerate(points):
                    mask[i] = boundary_geom.contains(point)
                
                mask = mask.reshape(cropped_data.shape[-2:])
                cropped_data = cropped_data.where(mask)
                
                valid_points = np.sum(mask)
                print(f"Masked: {valid_points}/{mask.size} points within India ({valid_points/mask.size*100:.1f}%)")
                
            except Exception as e:
                print(f"Shapefile error: {e}. Using bounds only.")
        
        return cropped_data
    
    else:
        print(f"Unknown region: {region}. Using world.")
        return data_array

def setup_map_projection(ax, region='world', shapefile_path=None, region_bounds=None):

    if region == 'world':
        ax.set_global()
        ax.add_feature(cfeature.COASTLINE, linewidth=0.5)

Plot the data side by side

def plot_side_by_side(ds_2019, ds_2020, variable_name, month, lp, up, region='world',
                      shapefile_path=None, region_bounds=None):
   
    # Select data for the specific month
    data_2019 = ds_2019[variable_name].sel(time=ds_2019.time.dt.month == month)
    data_2020 = ds_2020[variable_name].sel(time=ds_2020.time.dt.month == month)
    
    # If multiple time steps for the same month, take the mean
    if len(data_2019.time) > 1:
        data_2019 = data_2019.mean(dim='time')
    else:
        data_2019 = data_2019.squeeze()
        
    if len(data_2020.time) > 1:
        data_2020 = data_2020.mean(dim='time')
    else:
        data_2020 = data_2020.squeeze()
    
    # Prepare data for region
    data_2019 = prepare_data_for_region(data_2019, region, shapefile_path, region_bounds)
    data_2020 = prepare_data_for_region(data_2020, region, shapefile_path, region_bounds)
    
    # Calculate percentiles for filtering
    print(f"Calculating percentiles for {variable_name} over {region}...")
    p10_2019, p90_2019 = calculate_percentiles(data_2019, lp, up)
    p10_2020, p90_2020 = calculate_percentiles(data_2020, lp, up)
    
    # Filter data by percentiles 
    filtered_2019 = filter_data_by_percentiles(data_2019, p10_2019, p90_2019)
    filtered_2020 = filter_data_by_percentiles(data_2020, p10_2020, p90_2020)
    # print(filtered_2019.shape, filtered_2019.min(), filtered_2019.max())
    # print(filtered_2020.shape, filtered_2020.min(), filtered_2020.max())
    
    # Create figure with side-by-side subplots
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(16, 8),
                                   subplot_kw={'projection': ccrs.PlateCarree()})
    
    # Setup both map axes
    setup_map_projection(ax1, region, shapefile_path, region_bounds)
    setup_map_projection(ax2, region, shapefile_path, region_bounds)
    
    # Use the same colormap and scale for both plots
    vmin = min(filtered_2019.min().values, filtered_2020.min().values)
    vmax = max(filtered_2019.max().values, filtered_2020.max().values)
    
    # Plot 2019 data
    im1 = filtered_2019.plot(ax=ax1, transform=ccrs.PlateCarree(), 
                            cmap='viridis', add_colorbar=False, 
                            alpha=0.8, vmin=vmin, vmax=vmax)
    region_title = region.capitalize()
    ax1.set_title(f'{region_title} 2019 {variable_name} - Month {month}', 
                  fontsize=12, fontweight='bold')
    
    # Plot 2020 data
    im2 = filtered_2020.plot(ax=ax2, transform=ccrs.PlateCarree(), 
                            cmap='viridis', add_colorbar=False, 
                            alpha=0.8, vmin=vmin, vmax=vmax)
    ax2.set_title(f'{region_title} 2020 {variable_name} - Month {month}', 
                  fontsize=12, fontweight='bold')
    
    # Add colorbars
    cbar1 = plt.colorbar(im1, ax=ax1, shrink=0.6, pad=0.05)
    cbar1.set_label(f'{variable_name}', fontsize=10)
    
    cbar2 = plt.colorbar(im2, ax=ax2, shrink=0.6, pad=0.05)
    cbar2.set_label(f'{variable_name}', fontsize=10)
    
    plt.tight_layout()
    
    # Add main title
    month_names = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun',
                   'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec']
    fig.suptitle(f'{region_title} {variable_name} Comparison: {month_names[month-1]} 2019 vs 2020', 
                 fontsize=14, fontweight='bold', y=0.95)
    
    plt.show()
    
    print(f"{region_title} data range - Min: {vmin:.2e}, Max: {vmax:.2e}")    
    return fig, filtered_2019, filtered_2020

Calculate difference

def calculate_difference(ds_2019, ds_2020, variable_name, month, lp, up, region='world',
                        shapefile_path=None, region_bounds=None):
    
    # Select data for the specific month
    data_2019 = ds_2019[variable_name].sel(time=ds_2019.time.dt.month == month)
    data_2020 = ds_2020[variable_name].sel(time=ds_2020.time.dt.month == month)
    
    # If multiple time steps for the same month, take the mean
    if len(data_2019.time) > 1:
        data_2019 = data_2019.mean(dim='time')
    else:
        data_2019 = data_2019.squeeze()
        
    if len(data_2020.time) > 1:
        data_2020 = data_2020.mean(dim='time')
    else:
        data_2020 = data_2020.squeeze()
    
    # Prepare data for region
    data_2019 = prepare_data_for_region(data_2019, region, shapefile_path, region_bounds)
    data_2020 = prepare_data_for_region(data_2020, region, shapefile_path, region_bounds)
    
    # Calculate difference (2020 - 2019)
    difference = data_2020 - data_2019
    
    # Calculate percentiles for filtering the difference
    print(f"Calculating percentiles for {region} difference data...")
    p10_diff, p90_diff = calculate_percentiles(difference, lp, up)
    
    # Filter difference by percentiles
    filtered_difference = filter_data_by_percentiles(difference, p10_diff, p90_diff)
    
    return difference, filtered_difference

Complete flow

def complete_analysis(file_2019, file_2020, variable_name, month, lp, up, region='world',
                     shapefile_path=None, region_bounds=None):
    
    print(f"EMISSION ANALYSIS: {region.upper()} - {variable_name} - Month {month}")
    
    # Step 1: Read files
    print("\n1: Reading files")
    ds_2019, ds_2020 = read_files(file_2019, file_2020)
    
    # Check if variable exists
    # if variable_name not in ds_2019.data_vars:
    #     print(f"Error: Variable '{variable_name}' not found in datasets")
    #     print(f"Available variables: {list(ds_2019.data_vars)}")
    #     return None, None
    
    # Step 2: Plot side by side
    print(f"\n2: Plotting {region} {variable_name} side by side")
    fig1, filtered_2019, filtered_2020 = plot_side_by_side(ds_2019, ds_2020, 
                                                          variable_name, month, lp, up, region,
                                                          shapefile_path, region_bounds)
    
    # Step 3: Calculate difference
    print(f"\n3: Calculating {region} difference")
    difference, filtered_difference = calculate_difference(ds_2019, ds_2020, 
                                                          variable_name, month, lp ,up, region,
                                                          shapefile_path, region_bounds)
    
    # Step 4: Plot difference
    print(f"\n4: Plotting {region} difference")
    fig2 = plot_difference(difference, filtered_difference, 
                          variable_name, month, region, shapefile_path, 
                          region_bounds)
    
    print(f"\n{region.capitalize()} analysis complete!")
    return fig1, fig2, difference
